# Multiple curl commands for load testing
# Lines starting with # are comments

# GET request to users endpoint
curl http://127.0.0.1:9191/echo

# POST request with JSON data
curl -X POST http://127.0.0.1:9191/api/users -H "Content-Type: application/json" -d '{"test":"data"}'

# GET with custom headers
curl -H "User-Agent: gurl-test" http://127.0.0.1:9191/fast

# Delay endpoint
curl http://127.0.0.1:9191/slow
